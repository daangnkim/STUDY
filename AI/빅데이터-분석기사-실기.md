import pandas as pd

# 데이터 로드
df = pd.read_csv("type1_data1.csv")

# 1. 문자 자료형(object) 컬럼 삭제
df = df.select_dtypes(exclude='object')

# 2. 숫자 자료형 컬럼의 결측치를 0으로 대체
df = df.fillna(0)

# 3. 각 행의 합이 3000보다 큰 값의 개수 구하기
result = (df.sum(axis=1) > 3000).sum()
print(result)
